PROJECT(test)
CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
ENABLE_LANGUAGE(C)
IF(NOT "${DISABLE_OPENMP}")
  FIND_PACKAGE(OpenMP)
ENDIF(NOT "${DISABLE_OPENMP}")
IF(OPENMP_FOUND)
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} \
${OpenMP_CXX_FLAGS}")
ENDIF(OPENMP_FOUND)
SET(SRCS "test.c")
ADD_EXECUTABLE(test ${SRCS})
